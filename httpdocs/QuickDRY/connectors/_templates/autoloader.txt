<?php
/**
 * @param string $class
 */
function [[autoloader_class]](string $class) {
    $class_map = [
        [[mod_map]]
    ];

    if(!isset($class_map[$class])) {
      return;
    }

    $file = __DIR__ . '/../' . $class_map[$class];

    if (file_exists($file)) {
        require_once $file;
    }
}


spl_autoload_register('[[autoloader_class]]');
